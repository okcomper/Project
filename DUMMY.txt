CREATE TABLE USER(
    UNIQUE_ID INT(100) UNSIGNED AUTO_INCREMENT NOT NULL PRIMARY KEY,
    ID VARCHAR(15) NOT NULL,
    PW VARCHAR(15) NOT NULL,
    USER_NAME VARCHAR(15) NOT NULL,
    AGE INT(10) NOT NULL,
    USER_ADDRESS VARCHAR(50) NOT NULL
);

CREATE TABLE PRODUCT(
    PRODUCT_ID INT(100) UNSIGNED AUTO_INCREMENT NOT NULL PRIMARY KEY,
    UNIQUE_ID INT(100) UNSIGNED,
    PRODUCT_NAME VARCHAR(100) NOT NULL,
    PRODUCT_COST INT(10) NOT NULL,
    PRODUCT_DATE TIMESTAMP DEFAULT NOW(),
    PRODUCT_DESCRIPTION VARCHAR(100) NOT NULL,
    FOREIGN KEY (UNIQUE_ID) REFERENCES USER (UNIQUE_ID)
);
